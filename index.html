import pandas as pd import matplotlib.pyplot as plt # 1. สร้างข้อมูลจำลอง (นำไปแทนที่ด้วยข้อมูลจริงของคุณได้เลย) data = { 'Stock': ['PTT', 'AOT', 'CPALL', 'ADVANC', 'DELTA', 'SCC'], 'Percent': [-2.5, 5.2, -4.1, 1.8, 10.5, -1.0] } df = pd.DataFrame(data) # ========================================== # ส่วนที่ 1: การแก้ปัญหาเรียงกราฟจากยาวไปหาน้อย # ========================================== # เรียงลำดับจากเปอร์เซ็นต์มากไปน้อย df = df.sort_values(by='Percent', ascending=False) # ========================================== # ส่วนที่ 2: การตั้งค่ากราฟและการพล็อต # ========================================== fig, ax = plt.subplots(figsize=(10, 6)) # สร้างกราฟแท่ง (กำหนดให้ค่าบวกเป็นสีเขียว ค่าลบเป็นสีแดง) colors = ['#2ca02c' if val > 0 else '#d62728' for val in df['Percent']] bars = ax.bar(df['Stock'], df['Percent'], color=colors) # สร้างเส้นฐานที่ 0 (ช่วยให้เห็นชัดเจนว่าจุดไหนคือเส้นแบ่งบวกลบ) ax.axhline(0, color='black', linewidth=1) # ========================================== # ส่วนที่ 3: การขยับชื่อหุ้นลงไม่ให้ทับเปอร์เซ็นต์ที่ติดลบ # ========================================== # เพิ่มระยะห่าง (pad) ของชื่อหุ้นในแกน X ให้ถอยลงมาด้านล่าง ax.tick_params(axis='x', pad=15) # ========================================== # ส่วนเสริม: การใส่ตัวเลขเปอร์เซ็นต์ไม่ให้ทับกราฟ # ========================================== for bar in bars: yval = bar.get_height() # ถ้าค่าบวกให้ตัวเลขอยู่เหนือแท่งกราฟ, ถ้าค่าลบให้อยู่ใต้แท่งกราฟ offset = 5 if yval >= 0 else -15 ax.annotate(f'{yval}%', xy=(bar.get_x() + bar.get_width() / 2, yval), xytext=(0, offset), textcoords="offset points", ha='center', va='bottom' if yval >= 0 else 'top', fontsize=10) # ตกแต่งความสวยงามของกราฟ plt.title('Stock Percentage Change', fontsize=14, pad=20) plt.xlabel('Stock Name', fontsize=12, labelpad=10) plt.ylabel('Percentage (%)', fontsize=12) plt.grid(axis='y', linestyle='--', alpha=0.4) # จัดระเบียบเลย์เอาต์ไม่ให้ขอบกราฟโดนตัด plt.tight_layout() # แสดงผลกราฟ plt.show()
