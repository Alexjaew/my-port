function renderHeroBgChart() { const svg = document.getElementById('heroBgChart'); if (!svg) return; const data = state.history.filter(h => h['Date'] && h['มูลค่าพอร์ต'] != null); if (data.length < 2) { svg.innerHTML = ''; return; } const W = 400, H = 200; const values = data.map(d => Number(d['มูลค่าพอร์ต'])); // 1. ตั้งค่าฐาน (เช่น เงินต้น) const baseValue = Number(state.principal) || values[0]; // 2. คำนวณขอบเขตสูงสุดเพื่อไม่ให้กราฟล้น const maxVal = Math.max(...values, baseValue); const diff = Math.max(Math.abs(maxVal - baseValue), 1); // ระยะห่างจากฐานถึงจุดสูงสุด // 3. กำหนดตำแหน่ง Y โดยให้ baseValue อยู่ที่ระดับเส้นสีขาวเสมอ // สมมติว่าเส้นสีขาวอยู่ล่างๆ ของกราฟ (เช่น 80% ของความสูง) const baseLineY = H * 0.8; const scale = (H * 0.7) / diff; // พื้นที่การขยับของกราฟ const xStep = W / (data.length - 1); const points = values.map((v, i) => [i * xStep, baseLineY - (v - baseValue) * scale]); const pathD = points.map((p, i) => (i === 0 ? 'M' : 'L') + p[0].toFixed(1) + ',' + p[1].toFixed(1)).join(' '); const areaD = pathD + ` L${points[points.length-1][0].toFixed(1)},${baseLineY} L${points[0][0].toFixed(1)},${baseLineY} Z`; svg.innerHTML = ` `; }
